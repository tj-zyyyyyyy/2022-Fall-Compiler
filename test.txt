int add(int a,int b)
{
	return a+b;
}

int squaresum(int n)
{
	int sum;
	int i;
	sum=0;
	i=0;
	if(n<=0)
	{
		return 0;
	}
	while(i<=n)
	{
		sum=sum+i*i;
	}
	return sum;
}
